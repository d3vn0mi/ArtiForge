```python
# artiforge/detectors/sigma_models.py
"""Data models for Sigma rules."""

from __future__ import annotations

import re
from dataclasses import dataclass, field
from pathlib import Path


@dataclass
class LogSource:
    """Sigma logsource specification."""
    product: str | None = None
    service: str | None = None
    category: str | None = None


@dataclass
class SigmaRule:
    """A parsed Sigma detection rule."""
    title: str
    id: str | None
    logsource: LogSource
    detection: dict                   # raw detection block from YAML
    level: str                        # informational, low, medium, high, critical
    description: str
    tags: list[str] = field(default_factory=list)
    source_path: Path | None = None

    @property
    def mitre_ids(self) -> list[str]:
        """Extract MITRE ATT&CK technique IDs from tags."""
        pattern = re.compile(r"^attack\.t(\d+(?:\.\d+)*)$", re.IGNORECASE)
        result = []
        for tag in self.tags:
            m = pattern.match(tag)
            if m:
                result.append(f"T{m.group(1)}")
        return result
```

```python
# artiforge/detectors/sigma_loader.py
"""Load and validate Sigma YAML rule files."""

from __future__ import annotations

import re
import sys
from pathlib import Path

import yaml

from artiforge.detectors.sigma_models import LogSource, SigmaRule

# Patterns that indicate unsupported Sigma features
_UNSUPPORTED_CONDITION = re.compile(
    r"\b(count|min|max|sum|avg|near)\s*\(", re.IGNORECASE
)
_UNSUPPORTED_MODIFIERS = {"|re", "|base64", "|base64offset", "|cidr"}


def load_sigma_rule(path: Path) -> SigmaRule | None:
    """Parse a single Sigma YAML file.

    Returns None (with a warning to stderr) if the rule is missing
    required fields or uses unsupported features.
    """
    try:
        raw = yaml.safe_load(path.read_text(encoding="utf-8"))
    except Exception as exc:
        print(f"  [sigma] skip {path.name}: YAML parse error: {exc}", file=sys.stderr)
        return None

    if not isinstance(raw, dict):
        return None

    # Required fields
    title = raw.get("title")
    detection = raw.get("detection")
    logsource_raw = raw.get("logsource", {})

    if not title:
        print(f"  [sigma] skip {path.name}: missing 'title'", file=sys.stderr)
        return None
    if not detection or not isinstance(detection, dict):
        print(f"  [sigma] skip {path.name}: missing 'detection'", file=sys.stderr)
        return None

    condition = detection.get("condition", "")
    if not condition:
        print(f"  [sigma] skip {path.name}: missing 'detection.condition'", file=sys.stderr)
        return None

    # Check for unsupported features
    if _UNSUPPORTED_CONDITION.search(condition):
        print(f"  [sigma] skip {path.name}: aggregation condition not supported", file=sys.stderr)
        return None

    # Check for unsupported modifiers in field names
    for key, block in detection.items():
        if key == "condition":
            continue
        if isinstance(block, dict):
            for field_name in block:
                for mod in _UNSUPPORTED_MODIFIERS:
                    if mod in field_name:
                        print(
                            f"  [sigma] skip {path.name}: "
                            f"unsupported modifier '{mod}' in field '{field_name}'",
                            file=sys.stderr,
                        )
                        return None

    logsource = LogSource(
        product=logsource_raw.get("product"),
        service=logsource_raw.get("service"),
        category=logsource_raw.get("category"),
    )

    return SigmaRule(
        title=title,
        id=raw.get("id"),
        logsource=logsource,
        detection=detection,
        level=raw.get("level", "medium"),
        description=raw.get("description", ""),
        tags=raw.get("tags", []),
        source_path=path,
    )


def load_sigma_dir(dir_path: Path) -> list[SigmaRule]:
    """Load all .yml files from a directory. Skips invalid/unsupported rules."""
    if not dir_path.is_dir():
        return []
    rules = []
    for p in sorted(dir_path.glob("*.yml")):
        rule = load_sigma_rule(p)
        if rule is not None:
            rules.append(rule)
    return rules
```

```python
# artiforge/detectors/sigma_evaluator.py
"""Lightweight Sigma rule evaluator.

Evaluates Sigma rules against GeneratedEvent lists without pySigma.
Supports: selections with field matchers, modifiers (contains, startswith,
endswith, all), wildcard values, list OR, simple conditions (and, or, not,
1 of, all of).
"""

from __future__ import annotations

import fnmatch
import re
from typing import Any

from artiforge.core.models import GeneratedEvent
from artiforge.detectors.sigma_models import LogSource, SigmaRule

# ── Logsource → channel/EID mapping ─────────────────────────────────────

_SERVICE_CHANNEL = {
    "sysmon": "Sysmon",
    "security": "Security",
    "system": "System",
    "powershell": "PowerShell",
    "application": "Application",
    "wmi": "WMI",
}

_CATEGORY_FILTER: dict[str, tuple[str, list[int]]] = {
    "process_creation":       ("Sysmon", [1]),

def _filter_by_logsource(
    events: list[GeneratedEvent],
    logsource: LogSource,
) -> list[GeneratedEvent]:
    """Filter events by logsource (channel + optional EID)."""
    # First, try category mapping
    if logsource.category:
        cat = logsource.category.lower()
        if logsource.service:
            # Category with explicit service
            if cat in _CATEGORY_FILTER:
                channel, eids = _CATEGORY_FILTER[cat]
                return [e for e in events if e.channel == channel and e.eid in eids]
        else:
            # Category without service — try Sysmon first, then Security
            if cat in _CATEGORY_FILTER:
                channel, eids = _CATEGORY_FILTER[cat]
                return [e for e in events if e.channel == channel and e.eid in eids]
            if cat in _CATEGORY_SECURITY:
                channel, eids = _CATEGORY_SECURITY[cat]
                return [e for e in events if e.channel == channel and e.eid in eids]

    # Service-only mapping
    if logsource.service:
        svc = logsource.service.lower()
        channel = _SERVICE_CHANNEL.get(svc)
        if channel:
            return [e for e in events if e.channel == channel]

    # No logsource filter — return all events
    return list(events)


# ── Field matching ───────────────────────────────────────────────────────

def _parse_field_spec(field_spec: str) -> tuple[str, list[str]]:
    """Parse 'FieldName|modifier1|modifier2' into (field_name, [modifiers])."""
    parts = field_spec.split("|")
    return parts[0], parts[1:]


def _match_value(event_value: str, pattern: str, modifiers: list[str]) -> bool:
    """Check if event_value matches pattern with modifiers."""
    ev_lower = event_value.lower()
    pat_lower = pattern.lower()

    if "contains" in modifiers:
        return pat_lower in ev_lower
    if "startswith" in modifiers:
        return ev_lower.startswith(pat_lower)
    if "endswith" in modifiers:
        return ev_lower.endswith(pat_lower)

    # Default: exact match or wildcard
    if "*" in pattern or "?" in pattern:
        return fnmatch.fnmatch(ev_lower, pat_lower)
    return ev_lower == pat_lower


def _match_field(event_data: dict[str, str], field_spec: str, values: Any) -> bool:
    """Check if an event's data matches a single field specification."""
    field_name, modifiers = _parse_field_spec(field_spec)

    # Special case: EventID field matches against event_data or the EID directly
    event_value = str(event_data.get(field_name, ""))
    if not event_value and field_name == "EventID":
        return False

    # Normalize to list
    if not isinstance(values, list):
        values = [values]

    # Convert all values to strings
    str_values = [str(v) for v in values]

    if "all" in modifiers:
        # ALL values must match
        return all(_match_value(event_value, v, modifiers) for v in str_values)
    else:
        # ANY value matches (OR)
        return any(_match_value(event_value, v, modifiers) for v in str_values)


def _match_selection(event: GeneratedEvent, selection: dict) -> bool:
    """Check if an event matches all fields in a selection (AND logic)."""
    for field_spec, values in selection.items():
        if not _match_field(event.event_data, field_spec, values):
            return False
    return True


# ── Condition parsing ────────────────────────────────────────────────────

def _get_selections(detection: dict) -> dict[str, dict]:
    """Extract named selections from the detection block."""
    return {
        k: v for k, v in detection.items()
        if k != "condition" and isinstance(v, dict)
    }


def _eval_condition(
    condition: str,
    selections: dict[str, dict],
    event: GeneratedEvent,
) -> bool:
    """Evaluate a Sigma condition expression for a single event."""
    condition = condition.strip()

    # Handle "1 of <pattern>"
    m = re.match(r"^1 of (\w+)\*$", condition)
    if m:
        prefix = m.group(1)
        return any(
            _match_selection(event, sel)
            for name, sel in selections.items()
            if name.startswith(prefix)
        )

    # Handle "all of <pattern>"
    m = re.match(r"^all of (\w+)\*$", condition)
    if m:
        prefix = m.group(1)
        matching = [
            name for name, sel in selections.items()
            if name.startswith(prefix)
        ]
        if not matching:
            return False
        return all(
            _match_selection(event, selections[name])
            for name in matching
        )

    # Handle "A and not B"
    m = re.match(r"^(\w+)\s+and\s+not\s+(\w+)$", condition)
    if m:
        sel_name, filter_name = m.group(1), m.group(2)
        sel = selections.get(sel_name, {})
        flt = selections.get(filter_name, {})
        return _match_selection(event, sel) and not _match_selection(event, flt)

    # Handle "A or B"
    if " or " in condition and " and " not in condition:
        parts = [p.strip() for p in condition.split(" or ")]
        return any(
            _match_selection(event, selections.get(p, {}))
            for p in parts
        )

    # Handle "A and B"
    if " and " in condition and " not " not in condition:
        parts = [p.strip() for p in condition.split(" and ")]
        return all(
            _match_selection(event, selections.get(p, {}))
            for p in parts
        )

    # Handle "not A"
    m = re.match(r"^not\s+(\w+)$", condition)
    if m:
        sel = selections.get(m.group(1), {})
        return not _match_selection(event, sel)

    # Simple: single selection name
    sel = selections.get(condition, {})
    return _match_selection(event, sel)


# ── Public API ───────────────────────────────────────────────────────────

def evaluate_rule(
    rule: SigmaRule,
    events: list[GeneratedEvent],
) -> list[GeneratedEvent]:
    """Evaluate a Sigma rule against a list of events.

    Only evaluates attack events (phase_id != 0).
    Returns the list of matching events.
    """
    # Exclude noise
    attack_events = [e for e in events if e.phase_id != 0]

    # Filter by logsource
    candidates = _filter_by_logsource(attack_events, rule.logsource)

    # Get selections and condition
    selections = _get_selections(rule.detection)
    condition = rule.detection.get("condition", "")

    # Evaluate condition against each candidate
    return [e for e in candidates if _eval_condition(condition, selections, e)]
```
