**Goal:** Add Linux auditd support to ArtiForge — platform-aware host model, 7 auditd record type generators with compound aliases, raw `audit.log` exporter, and ECS NDJSON mappings for Auditbeat compatibility.

**Architecture:** A new `platform` field on `Host` controls routing. A new `linux_auditd.py` generator module handles all auditd record types. Events use `channel="Auditd"` and link related records via shared `audit_id` (timestamp:serial). A new `auditd_exporter.py` produces raw `audit.log` files, and `elastic.py` gains Linux-aware ECS mappings.

**Tech Stack:** Python 3.10+, Pydantic v2, pytest

# artiforge/generators/linux_auditd.py
"""Linux auditd record type generators.

Produces event_data dicts for auditd record types:
  SYSCALL (1300), EXECVE (1309), PATH (1302), SOCKADDR (1306),
  USER_AUTH (1100), USER_LOGIN (1101), CRED_ACQ (1103)
"""

from __future__ import annotations

import random
from typing import Any

from artiforge.core.models import Host, User


# ── EID aliases ──────────────────────────────────────────────────────────

_ALIASES: dict[str, int] = {
    "SYSCALL": 1300,
    "EXECVE": 1309,
    "PATH": 1302,
    "SOCKADDR": 1306,
    "USER_AUTH": 1100,
    "USER_LOGIN": 1101,
    "CRED_ACQ": 1103,
}


def resolve_eid(eid: int | str) -> int:
    """Resolve a string alias or pass through a numeric EID."""
    if isinstance(eid, str):
        if eid in _ALIASES:
            return _ALIASES[eid]
        try:
            return int(eid)
        except ValueError:
            raise ValueError(f"Unknown auditd record type alias: {eid!r}")
    return eid


# ── Helpers ──────────────────────────────────────────────────────────────

def _pid() -> str:
    return str(random.randint(1000, 65535))


def _inode() -> str:
    return str(random.randint(100000, 999999))


def _ses() -> str:
    return str(random.randint(1, 100))


def _auid(user: User | None) -> str:
    return str(user.rid if user else 1000)


def _uid(user: User | None) -> str:
    return str(user.rid if user else 0)


# ── Record type generators ──────────────────────────────────────────────

def eid_1300(fields: dict, host: Host, user: User | None, **_) -> dict:
    """SYSCALL — system call metadata."""
    exe = fields.get("exe", "/usr/bin/bash")
    comm = exe.rsplit("/", 1)[-1] if "/" in exe else exe
    return {
        "arch": fields.get("arch", "c000003e"),
        "syscall": str(fields.get("syscall", "59")),
        "success": fields.get("success", "yes"),
        "exit": str(fields.get("exit", "0")),
        "ppid": fields.get("ppid", _pid()),
        "pid": fields.get("pid", _pid()),
        "auid": fields.get("auid", _auid(user)),
        "uid": fields.get("uid", _uid(user)),
        "gid": fields.get("gid", _uid(user)),
        "euid": fields.get("euid", _uid(user)),
        "suid": fields.get("suid", _uid(user)),
        "fsuid": fields.get("fsuid", _uid(user)),
        "egid": fields.get("egid", _uid(user)),
        "sgid": fields.get("sgid", _uid(user)),
        "fsgid": fields.get("fsgid", _uid(user)),
        "tty": fields.get("tty", "pts0"),
        "ses": fields.get("ses", _ses()),
        "comm": fields.get("comm", f'"{comm}"'),
        "exe": fields.get("exe", exe),
        "key": fields.get("key", '"exec_monitor"'),
    }


def eid_1309(fields: dict, host: Host, user: User | None, **_) -> dict:
    """EXECVE — command execution arguments."""
    args = fields.get("args", ["bash"])
    if isinstance(args, str):
        args = [args]
    result = {"argc": str(len(args))}
    for i, arg in enumerate(args):
        result[f"a{i}"] = str(arg)
    return result


def eid_1302(fields: dict, host: Host, user: User | None, **_) -> dict:
    """PATH — file path accessed by a syscall."""
    return {
        "item": str(fields.get("item", "0")),
        "name": fields.get("name", "/usr/bin/bash"),
        "inode": fields.get("inode", _inode()),
        "dev": fields.get("dev", "08:01"),
        "mode": fields.get("mode", "0100755"),
        "ouid": fields.get("ouid", "0"),
        "ogid": fields.get("ogid", "0"),
        "rdev": fields.get("rdev", "00:00"),
        "nametype": fields.get("nametype", "NORMAL"),
    }


def eid_1306(fields: dict, host: Host, user: User | None, **_) -> dict:
    """SOCKADDR — network socket address."""
    return {
        "saddr": fields.get("saddr", "02000050AC100A32"),
        "family": fields.get("family", "inet"),
        "laddr": fields.get("laddr", host.ip),
        "lport": str(fields.get("lport", str(random.randint(1024, 65535)))),
        "addr": fields.get("addr", "10.10.10.10"),
        "port": str(fields.get("port", "443")),
    }


def eid_1100(fields: dict, host: Host, user: User | None, **_) -> dict:
    """USER_AUTH — PAM authentication."""
    acct = fields.get("acct", user.username if user else "root")
    exe = fields.get("exe", "/usr/bin/sudo")
    hostname = fields.get("hostname", "?")
    addr = fields.get("addr", "?")
    terminal = fields.get("terminal", "/dev/pts/0")
    res = fields.get("res", "success")
    return {
        "pid": fields.get("pid", _pid()),
        "uid": fields.get("uid", _uid(user)),
        "auid": fields.get("auid", _auid(user)),
        "ses": fields.get("ses", _ses()),
        "msg": (
            f'op=PAM:authentication acct="{acct}" '
            f'exe="{exe}" hostname={hostname} addr={addr} '
            f'terminal={terminal} res={res}'
        ),
    }


def eid_1101(fields: dict, host: Host, user: User | None, **_) -> dict:
    """USER_LOGIN — login session."""
    uid = _uid(user)
    exe = fields.get("exe", "/usr/sbin/sshd")
    hostname = fields.get("hostname", "10.10.10.10")
    addr = fields.get("addr", hostname)
    terminal = fields.get("terminal", "ssh")
    res = fields.get("res", "success")
    return {
        "pid": fields.get("pid", _pid()),
        "uid": fields.get("uid", uid),
        "auid": fields.get("auid", _auid(user)),
        "ses": fields.get("ses", _ses()),
        "msg": (
            f'op=login id={uid} '
            f'exe="{exe}" hostname={hostname} addr={addr} '
            f'terminal={terminal} res={res}'
        ),
    }


def eid_1103(fields: dict, host: Host, user: User | None, **_) -> dict:
    """CRED_ACQ — credential acquisition (su, sudo)."""
    acct = fields.get("acct", user.username if user else "root")
    exe = fields.get("exe", "/usr/bin/sudo")
    hostname = fields.get("hostname", "?")
    addr = fields.get("addr", "?")
    terminal = fields.get("terminal", "/dev/pts/0")
    res = fields.get("res", "success")
    return {
        "pid": fields.get("pid", _pid()),
        "uid": fields.get("uid", _uid(user)),
        "auid": fields.get("auid", _auid(user)),
        "ses": fields.get("ses", _ses()),
        "msg": (
            f'op=PAM:setcred acct="{acct}" '
            f'exe="{exe}" hostname={hostname} addr={addr} '
            f'terminal={terminal} res={res}'
        ),
    }


# ── Dispatcher ───────────────────────────────────────────────────────────

_GENERATORS = {
    1300: eid_1300,
    1309: eid_1309,
    1302: eid_1302,
    1306: eid_1306,
    1100: eid_1100,
    1101: eid_1101,
    1103: eid_1103,
}


def generate(eid: int, fields: dict, host: Host, user: User | None,
             spec: Any, timestamp: Any, ctx: Any = None,
             session_label: str = "default",
             process_label: str = "default") -> dict:
    """Generate an auditd record's event_data dict."""
    fn = _GENERATORS.get(eid)
    if fn is None:
        raise ValueError(f"Auditd EID {eid} not implemented.")
    return fn(fields=fields, host=host, user=user, spec=spec, timestamp=timestamp)
```

# artiforge/exporters/auditd_exporter.py
"""Raw audit.log exporter for Linux auditd events.

Produces one audit.log file per Linux host in the native key=value format.
Windows events are silently skipped.
"""

from __future__ import annotations

from collections import defaultdict
from pathlib import Path

from artiforge.core.models import ArtifactBundle, GeneratedEvent

# Map EID back to record type name
_RECORD_TYPES = {
    1300: "SYSCALL",
    1309: "EXECVE",
    1302: "PATH",
    1306: "SOCKADDR",
    1100: "USER_AUTH",
    1101: "USER_LOGIN",
    1103: "CRED_ACQ",
}


def _format_record(ev: GeneratedEvent) -> str:
    """Format a single auditd event as a raw audit.log line."""
    record_type = _RECORD_TYPES.get(ev.eid, f"UNKNOWN[{ev.eid}]")
    unix_ts = ev.timestamp.timestamp()
    serial = ev.record_id

    # Build key=value pairs from event_data
    kv_pairs = " ".join(f"{k}={v}" for k, v in ev.event_data.items())

    return f"type={record_type} msg=audit({unix_ts:.3f}:{serial}): {kv_pairs}"


def export(bundle: ArtifactBundle, output_dir: Path) -> list[Path]:
    """Write one audit.log per Linux host. Skip non-Auditd events."""
    output_dir.mkdir(parents=True, exist_ok=True)

    groups: dict[str, list[GeneratedEvent]] = defaultdict(list)
    for ev in bundle.events:
        if ev.channel == "Auditd":
            groups[ev.host].append(ev)

    written: list[Path] = []

    for host, events in sorted(groups.items()):
        lines = []
        for ev in sorted(events, key=lambda e: e.timestamp):
            lines.append(_format_record(ev))

        out_path = output_dir / f"{host}_audit.log"
        out_path.write_text("\n".join(lines) + "\n", encoding="utf-8")
        written.append(out_path)

    return written
```
