```python
# artiforge/generators/forensic_artifacts.py
"""Forensic artifact orchestrator.

Scans a generated bundle for process execution events (Sysmon 1, Security 4688)
and produces correlated Prefetch, Amcache, and $MFT artifacts.
"""

from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime
from pathlib import Path

from artiforge.core.models import ArtifactBundle


@dataclass
class ProcessInfo:
    """Metadata for a unique process execution, shared across generators."""
    image_path: str
    image_name: str
    parent_dir: str
    first_run: datetime
    run_count: int
    hashes: dict[str, str]
    file_version: str
    original_filename: str
    company: str
    host: str


def _parse_hashes(hash_str: str) -> dict[str, str]:
    """Parse 'MD5=AABB,SHA256=CCDD' into {'MD5': 'AABB', 'SHA256': 'CCDD'}."""
    result = {}
    if not hash_str:
        return result
    for part in hash_str.split(","):
        if "=" in part:
            k, v = part.split("=", 1)
            result[k.strip()] = v.strip()
    return result


def collect_process_info(bundle: ArtifactBundle) -> list[ProcessInfo]:
    """Extract unique process executions from attack events.

    Scans Sysmon 1 and Security 4688 events (phase_id != 0),
    deduplicates by (host, image_path), aggregates run counts,
    and tracks earliest execution timestamp.
    """
    seen: dict[tuple[str, str], ProcessInfo] = {}

    for ev in bundle.events:
        if ev.phase_id == 0:
            continue

        image = None
        if ev.channel == "Sysmon" and ev.eid == 1:
            image = ev.event_data.get("Image", "")
        elif ev.channel == "Security" and ev.eid == 4688:
            image = ev.event_data.get("NewProcessName", "")

        if not image:
            continue

        key = (ev.host, image)
        if key in seen:
            seen[key].run_count += 1
            if ev.timestamp < seen[key].first_run:
                seen[key].first_run = ev.timestamp
        else:
            name = image.rsplit("\\", 1)[-1] if "\\" in image else image
            parent = image.rsplit("\\", 1)[0] if "\\" in image else ""
            seen[key] = ProcessInfo(
                image_path=image,
                image_name=name,
                parent_dir=parent,
                first_run=ev.timestamp,
                run_count=1,
                hashes=_parse_hashes(ev.event_data.get("Hashes", "")),
                file_version=ev.event_data.get("FileVersion", ""),
                original_filename=ev.event_data.get("OriginalFileName", name),
                company=ev.event_data.get("Company", ""),
                host=ev.host,
            )

    return list(seen.values())


def generate(bundle: ArtifactBundle, output_dir: Path) -> list[Path]:
    """Generate all forensic artifacts. Returns list of written paths."""
    from artiforge.generators.prefetch import generate_prefetch
    from artiforge.generators.amcache import generate_amcache
    from artiforge.generators.mft import generate_mft

    infos = collect_process_info(bundle)
    if not infos:
        return []

    # Group by host
    by_host: dict[str, list[ProcessInfo]] = {}
    for info in infos:
        by_host.setdefault(info.host, []).append(info)

    written: list[Path] = []

    for host, host_infos in sorted(by_host.items()):
        host_dir = output_dir / host

        # Prefetch — one .pf per process
        pf_dir = host_dir / "prefetch"
        pf_dir.mkdir(parents=True, exist_ok=True)
        for info in host_infos:
            pf_path = generate_prefetch(info, pf_dir)
            written.append(pf_path)

        # Amcache — one JSON per host
        ac_path = generate_amcache(host_infos, host_dir)
        written.append(ac_path)

        # $MFT — one JSON per host
        mft_path = generate_mft(host_infos, host_dir)
        written.append(mft_path)

    return written
```

```python
# artiforge/generators/prefetch.py
"""Windows Prefetch (.pf) binary file generator.

Produces Prefetch version 30 (Windows 10+) files with correct headers,
executable name, run count, and timestamps.
"""

from __future__ import annotations

import struct
import zlib
from datetime import datetime, timezone
from pathlib import Path

from artiforge.generators.forensic_artifacts import ProcessInfo

# FILETIME epoch diff (same as evtxforge)
_FILETIME_EPOCH_DIFF = 116444736000000000


def _to_filetime(dt: datetime) -> int:
    if dt.tzinfo is None:
        dt = dt.replace(tzinfo=timezone.utc)
    unix_seconds = int(dt.timestamp())
    microseconds = dt.microsecond
    ticks = unix_seconds * 10_000_000 + microseconds * 10
    return ticks + _FILETIME_EPOCH_DIFF


def prefetch_hash(path: str) -> int:
    """Compute a Prefetch-style hash from the executable path.

    Uses CRC32 of the uppercase path as a deterministic approximation
    of the real Windows Prefetch hash algorithm.
    """
    return zlib.crc32(path.upper().encode("utf-16-le")) & 0xFFFFFFFF


def generate_prefetch(info: ProcessInfo, output_dir: Path) -> Path:
    """Generate a single Prefetch .pf file for a process."""
    pf_hash = prefetch_hash(info.image_path)
    filename = f"{info.image_name.upper()}-{pf_hash:08X}.pf"
    out_path = output_dir / filename

    # Build 184-byte header
    buf = bytearray(184)

    # Version 30
    struct.pack_into("<I", buf, 0, 30)
    # Magic: MAM\x04
    buf[4:8] = b"MAM\x04"
    # Unknown field
    struct.pack_into("<I", buf, 8, 0x11)
    # File size (just header for now)
    struct.pack_into("<I", buf, 12, 184)

    # Executable name — UTF-16LE, padded to 60 chars
    exe_upper = info.image_name.upper()
    name_encoded = exe_upper.encode("utf-16-le")
    name_field = name_encoded[:120]  # max 60 chars * 2 bytes
    name_field = name_field.ljust(120, b"\x00")
    buf[16:136] = name_field

    # Prefetch hash
    struct.pack_into("<I", buf, 136, pf_hash)

    # Last run time (FILETIME) at offset 144
    struct.pack_into("<Q", buf, 144, _to_filetime(info.first_run))

    # Previous run times (3 slots at offsets 152, 160, 168) — zeroed
    # offset 152-175 already zero

    # Run count at offset 176
    struct.pack_into("<I", buf, 176, info.run_count)

    out_path.write_bytes(bytes(buf))
    return out_path
```

```python
# artiforge/generators/amcache.py
"""Amcache JSON artifact generator.

Produces amcache_entries.json with the same fields analysts extract
from Amcache.hve using AmcacheParser.
"""

from __future__ import annotations

import json
import hashlib
from pathlib import Path

from artiforge.generators.forensic_artifacts import ProcessInfo


def _generate_sha1(path: str) -> str:
    """Generate a deterministic fake SHA1 from the path."""
    return hashlib.sha1(path.encode("utf-8")).hexdigest().upper()


def generate_amcache(infos: list[ProcessInfo], output_dir: Path) -> Path:
    """Generate amcache_entries.json for a single host."""
    output_dir.mkdir(parents=True, exist_ok=True)

    entries = []
    for info in infos:
        sha1 = info.hashes.get("SHA1", _generate_sha1(info.image_path))
        entry = {
            "full_path": info.image_path,
            "sha1": sha1,
            "file_size": 0,
            "first_run": info.first_run.strftime("%Y-%m-%dT%H:%M:%SZ"),
            "last_modified": info.first_run.strftime("%Y-%m-%dT%H:%M:%SZ"),
            "publisher": info.company,
            "file_version": info.file_version,
            "pe_header_checksum": "0x00000000",
            "original_filename": info.original_filename,
        }
        entries.append(entry)

    out_path = output_dir / "amcache_entries.json"
    out_path.write_text(json.dumps(entries, indent=2) + "\n", encoding="utf-8")
    return out_path
```

```python
# artiforge/generators/mft.py
"""$MFT JSON artifact generator.

Produces mft_entries.json with the same fields analysts extract
from $MFT using MFTECmd.
"""

from __future__ import annotations

import json
import zlib
from datetime import timedelta
from pathlib import Path

from artiforge.generators.forensic_artifacts import ProcessInfo


def _record_number(path: str) -> int:
    """Deterministic MFT record number from path."""
    return (zlib.crc32(path.encode("utf-8")) & 0x7FFFFFFF) % 10000000 + 100000


def generate_mft(infos: list[ProcessInfo], output_dir: Path) -> Path:
    """Generate mft_entries.json for a single host."""
    output_dir.mkdir(parents=True, exist_ok=True)

    entries = []
    for info in infos:
        # Created = a few seconds before first execution
        created = info.first_run - timedelta(seconds=5)
        accessed = info.first_run

        entry = {
            "record_number": _record_number(info.image_path),
            "filename": info.image_name,
            "parent_directory": info.parent_dir,
            "file_size": 0,
            "created": created.strftime("%Y-%m-%dT%H:%M:%SZ"),
            "modified": created.strftime("%Y-%m-%dT%H:%M:%SZ"),
            "accessed": accessed.strftime("%Y-%m-%dT%H:%M:%SZ"),
            "entry_modified": created.strftime("%Y-%m-%dT%H:%M:%SZ"),
            "is_directory": False,
            "in_use": True,
        }
        entries.append(entry)

    out_path = output_dir / "mft_entries.json"
    out_path.write_text(json.dumps(entries, indent=2) + "\n", encoding="utf-8")
    return out_path
```
