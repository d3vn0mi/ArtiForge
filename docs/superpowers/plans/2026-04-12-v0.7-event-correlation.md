> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an event correlation engine so that related events share identifiers (LogonId, ProcessGuid, etc.), expand event coverage with 16 new EIDs, and add correlation-aware validation checks.

**Architecture:** A `CorrelationContext` class scoped per (phase, host) is passed into every generator. Producer EIDs (4624, Sysmon 1) write identifiers to the context; consumer EIDs read them back. Three-tier field precedence: YAML fields > correlation context > random default. New `session` and `process` labels on `EventSpec` support multiple concurrent sessions/processes.

**Tech Stack:** Python 3.10+, Pydantic v2, pytest, Click CLI

**Spec:** `docs/superpowers/specs/2026-04-12-v0.7-event-correlation-design.md`

```python
# artiforge/core/correlation.py
"""Event correlation context — shared identifiers across related events."""

from __future__ import annotations

from dataclasses import dataclass

from artiforge.core.models import Host


@dataclass
class SessionState:
    """Tracks a single logon session."""
    logon_id: str
    logon_guid: str
    user: str


@dataclass
class ProcessState:
    """Tracks a single process lifecycle."""
    process_guid: str
    process_id: str
    image: str
    parent_guid: str | None = None
    parent_id: str | None = None
    parent_image: str | None = None


class CorrelationContext:
    """Per-(phase, host) state shared across event generators.

    Tracks logon sessions and process chains so that related events
    can share identifiers (LogonId, ProcessGuid, etc.).
    """

    def __init__(self, host: Host) -> None:
        self.host = host
        self._sessions: dict[str, SessionState] = {}
        self._processes: dict[str, ProcessState] = {}
        self._current_session_label: str | None = None
        self._current_process_label: str | None = None

    # ── Session management ───────────────────────────────────────────────

    def register_session(
        self,
        logon_id: str,
        logon_guid: str,
        user: str,
        label: str = "default",
    ) -> SessionState:
        """Register a new logon session. Becomes the current session."""
        state = SessionState(logon_id=logon_id, logon_guid=logon_guid, user=user)
        self._sessions[label] = state
        self._current_session_label = label
        return state

    def get_session(self, label: str = "default") -> SessionState | None:
        """Retrieve a session by label."""
        return self._sessions.get(label)

    @property
    def current_session(self) -> SessionState | None:
        """The most recently registered session."""
        if self._current_session_label is None:
            return None
        return self._sessions.get(self._current_session_label)

    # ── Process management ───────────────────────────────────────────────

    def register_process(
        self,
        process_guid: str,
        process_id: str,
        image: str,
        label: str = "default",
    ) -> ProcessState:
        """Register a new process. Sets parent from current process. Becomes current."""
        current = self.current_process
        state = ProcessState(
            process_guid=process_guid,
            process_id=process_id,
            image=image,
            parent_guid=current.process_guid if current else None,
            parent_id=current.process_id if current else None,
            parent_image=current.image if current else None,
        )
        self._processes[label] = state
        self._current_process_label = label
        return state

    def get_process(self, label: str = "default") -> ProcessState | None:
        """Retrieve a process by label."""
        return self._processes.get(label)

    @property
    def current_process(self) -> ProcessState | None:
        """The most recently registered process."""
        if self._current_process_label is None:
            return None
        return self._processes.get(self._current_process_label)
```

```python
from artiforge.core.correlation import CorrelationContext

def _resolve(fields: dict, key: str, ctx: CorrelationContext | None,
             session_label: str, fallback_fn):
    """Three-tier field resolution: YAML > correlation > random default."""
    if key in fields:
        return fields[key]
    if ctx is not None:
        session = ctx.get_session(session_label)
        if session is not None:
            if key in ("SubjectLogonId", "TargetLogonId"):
                return session.logon_id
            if key == "LogonGuid":
                return session.logon_guid
    return fallback_fn()
```

```python
def _resolve_process(fields: dict, ctx: CorrelationContext | None,
                     process_label: str) -> tuple[str, str]:
    """Resolve ProcessGuid/ProcessId from fields, context, or random."""
    process_guid = fields.get("ProcessGuid")
    process_id = fields.get("ProcessId")
    if ctx is not None and process_guid is None:
        proc = ctx.get_process(process_label)
        if proc is not None:
            process_guid = proc.process_guid
            process_id = proc.process_id
    return process_guid or _guid(), process_id or _pid()
```
